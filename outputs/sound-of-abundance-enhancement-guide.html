import { motion, AnimatePresence } from 'framer-motion';
import { useEffect, useState } from 'react';

export function LoadingScreen() {
  const [show, setShow] = useState(false);

  useEffect(() => {
    if (!sessionStorage.getItem('soa-loaded')) {
      setShow(true);
    }
  }, []);

  const handleComplete = () => {
    sessionStorage.setItem('soa-loaded', '1');
    setShow(false);
  };

  if (!show) return null;

  return (
    <AnimatePresence>
      <motion.div
        initial={{ opacity: 1 }}
        exit={{ opacity: 0, transition: { duration: 0.4, delay: 1.2 } }}
        onAnimationComplete={handleComplete}
        style={{
          position: 'fixed', inset: 0,
          background: 'var(--bg)',
          display: 'flex', flexDirection: 'column',
          alignItems: 'center', justifyContent: 'center',
          zIndex: 9998,
        }}

        <motion.div

            fontFamily: 'var(--font-display)',

interface Props {
  children: React.ReactNode;
  delay?: number;
  y?: number;
  className?: string;
}

export function ScrollReveal({ children, delay = 0, y = 32, className }: Props) {
  const ref = useRef(null);
  const isInView = useInView(ref, { once: true, margin: '-80px' });

  return (
    <motion.div

      initial={{ opacity: 0, y }}

      transition={{ duration: 0.7, ease: [0.25, 0.1, 0.25, 1], delay }}
      className={className}
    >
      {children}
    </motion.div>
  );
}

"use client";
import { useEffect } from 'react';
import { motion, useMotionValue, useSpring } from 'framer-motion';

export function CustomCursor() {
  const mouseX = useMotionValue(0);
  const mouseY = useMotionValue(0);

  const springConfig = { stiffness: 200, damping: 20 };
  const cursorX = useSpring(mouseX, springConfig);
  const cursorY = useSpring(mouseY, springConfig);

  useEffect(() => {
    // Disable on touch/coarse-pointer devices
    if (window.matchMedia('(pointer: coarse)').matches) return;
    document.body.classList.add('has-custom-cursor');

    const move = (e: MouseEvent) => {
      mouseX.set(e.clientX);
      mouseY.set(e.clientY);
    };
    window.addEventListener('mousemove', move);
    return () => {
      window.removeEventListener('mousemove', move);
      document.body.classList.remove('has-custom-cursor');
    };
  }, [mouseX, mouseY]);

  return (
    <>
      <motion.div
        style={{
          position: 'fixed',
          left: cursorX,
          top: cursorY,
          width: 20, height: 20,
          border: '1px solid var(--accent)',
          borderRadius: '50%',
          transform: 'translate(-50%, -50%)',
          pointerEvents: 'none',
          zIndex: 9999,
          transition: 'width 0.2s, height 0.2s, border-color 0.2s',
        }}
        className="cursor-ring"
      />
      <motion.div
        style={{
          position: 'fixed',
          left: mouseX,
          top: mouseY,
          width: 4, height: 4,
          background: 'var(--accent)',
          borderRadius: '50%',
          transform: 'translate(-50%, -50%)',
          pointerEvents: 'none',
          zIndex: 10000,
        }}
      />
    </>
  );
}

function ConstellationCanvas({ active }: { active: boolean }) {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    if (!active || !canvasRef.current) return;
    const canvas = canvasRef.current;
    const ctx = canvas.getContext('2d')!;
    canvas.width = canvas.offsetWidth;
    canvas.height = canvas.offsetHeight;

    let progress = 0;
    const anim = () => {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      if (progress < 1) progress += 0.025;

      PLATFORMS.forEach((p, i) => {
        PLATFORMS.forEach((q, j) => {
          if (j > i) {

            const x2 = (parseFloat(q.x) / 100) * canvas.width;
            const y2 = (parseFloat(q.y) / 100) * canvas.height;

            ctx.beginPath();
            ctx.strokeStyle = `rgba(212,175,55,${0.12 * progress})`;
            ctx.lineWidth = 0.5;
            ctx.moveTo(x1, y1);
            ctx.lineTo(x2, y2);
            ctx.stroke();
          }
        });
      });
      if (progress < 1) requestAnimationFrame(anim);
    };
    anim();
  }, [active]);

  return (
    <canvas
      ref={canvasRef}
      style={{ position: 'absolute', inset: 0, width: '100%', height: '100%' }}
    />
  );
